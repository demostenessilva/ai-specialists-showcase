-- ACQUISITION: New Customer Acquisition Cost (CAC)
SELECT 
    DATE_TRUNC('month', c.first_purchase_date) AS cohort_month,
    m.marketing_spend_usd / COUNT(DISTINCT c.customer_id) AS cac_usd
FROM marts.dim_customers c
JOIN marts.fct_marketing_spend m 
    ON DATE_TRUNC('month', c.first_purchase_date) = m.month
WHERE c.first_purchase_date IS NOT NULL
GROUP BY 1, m.marketing_spend_usd
ORDER BY 1 DESC;

-- RETENTION: Repeat Purchase Rate (Cohort-based)
WITH first_purchases AS (
    SELECT customer_id, MIN(transaction_date) AS first_purchase_date
    FROM marts.fct_transactions 
    GROUP BY 1
),
second_purchases AS (
    SELECT t.customer_id
    FROM marts.fct_transactions t
    JOIN first_purchases fp 
        ON t.customer_id = fp.customer_id 
        AND t.transaction_date > fp.first_purchase_date
)
SELECT
    DATE_TRUNC('month', fp.first_purchase_date) AS cohort_month,
    COUNT(DISTINCT fp.customer_id) AS total_customers,
    COUNT(DISTINCT sp.customer_id) AS repeat_customers,
    ROUND(100.0 * COUNT(DISTINCT sp.customer_id) 
          / COUNT(DISTINCT fp.customer_id), 2) AS repeat_rate_pct
FROM first_purchases fp
LEFT JOIN second_purchases sp ON fp.customer_id = sp.customer_id
GROUP BY 1
ORDER BY 1 DESC;

-- REVENUE: GMV (Gross Merchandise Volume) by Category
SELECT
    DATE_TRUNC('month', d.date_actual) AS month,
    p.category_l1,
    SUM(t.net_amount_usd) AS gmv_usd
FROM marts.fct_transactions t
JOIN marts.dim_products p ON t.product_key = p.product_key
JOIN marts.dim_date d ON t.transaction_date_key = d.date_key
GROUP BY 1, 2
ORDER BY 1 DESC, 3 DESC;
